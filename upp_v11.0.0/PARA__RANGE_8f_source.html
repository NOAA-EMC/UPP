26 iwork1 = ( n2 - n1 + 1 ) / nprocs
27 iwork2 = mod( n2 - n1 + 1, nprocs )
28 ista = irank * iwork1 + n1 + min( irank, iwork2 )
29 iend = ista + iwork1 - 1

89 dalp = alpint(i,j,l) - alpint(i,j,l+1)
90 dz = zint(i,j,l) - zint(i,j,l+1)
91 zlcl(i,j) = max(d00, zint(i,j,l+1) + dz*dlplcl/dalp - zsfc)

94 20 CONTINUE

96 30 CONTINUE

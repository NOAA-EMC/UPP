850  subroutine calc_layerq(t, rh, pres, ept, nz, clouds)
851  IMPLICIT NONE
852  integer, intent(in) :: nz
853  real, intent(in) :: t(nz), rh(nz), pres(nz), ept(nz)
854  type(clouds_t), intent(inout) :: clouds
855

858  real :: totalq, testthetae, q_top, q_base, q_layer, delta_te, sum_rh
859  integer :: base_k, test_k, num_layers
860 
861  integer :: k, m, n, kk
862 
863  clouds%layerQ(:) = 0.0
864 
865  ! loop through each layer
866  lp_nlayers: do n = 1, clouds%nLayers
867  lp_k_inlayer: do k = clouds%topIdx(n), clouds%baseIdx(n)-1
868 
869  totalq = 0.0
870  testthetae = ept(k)
871 
872  ! get base layer k value (first more than 4K colder than at k)
873  base_k = clouds%baseIdx(n)
874  do m = k, nz-1
875  if((testthetae-ept(m)>4.) .and. (clouds%baseIdx(n)<=m))then
876  base_k = m - 1
877  exit
878  end if
879  end do
880 
881  ! calculate delta thetaE and deltaQ and deltaRH for the layer
882  lp_kk: do kk = k, base_k-1
883  q_top = mixing_ratio(t(kk), pres(kk))
884  q_base = mixing_ratio(t(kk+1), pres(kk+1))
885  q_layer = q_base - q_top
886 
887  ! convert Q from g/kg to g/m**3
888  q_layer = q_layer * (pres(kk)/(287.05 * t(kk)))
889 
890  if (q_layer < 0.0) q_layer = 0.0
891 
892  delta_te = testthetae - ept(kk+1)
893  test_k = kk + 1
894 
895  ! get the mean RH in the layer
896  num_layers = 0
897  sum_rh = 0.0
898  do m = k, test_k
899  num_layers = num_layers + 1
900  sum_rh = sum_rh + rh(m)
901  end do
902 
903  if (num_layers == 0) then
904  mean_rh = 0.0
905  else
906  mean_rh = sum_rh / num_layers
907  end if

909  te_map = dq_delta_te_map(delta_te)
910  rh_map = dq_rh_map(mean_rh)
911  adj_q = q_layer * te_map * rh_map

913  totalq = totalq + adj_q
914  end do lp_kk
915 
916  clouds%layerQ(k) = totalq
917  end do lp_k_inlayer
918  end do lp_nlayers

920  return

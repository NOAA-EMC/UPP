123  u2= u10(i,j) + (u(i,j,k1-1)-u10(i,j))*599.6/hz1 !found it between K1-1 and K1, then linear
124  v2= v10(i,j) + (v(i,j,k1-1)-v10(i,j))*599.6/hz1 !interpolate to get wind at 2000ft U2,V2
125  z2= fis(i,j)*gi + 609.6
126  ELSE !otherwise, keep on search upward
127  DO lp = k1-1,2,-1
128  dh=dh+(h(i,j,lp-1) - h(i,j,lp))
129  IF((dh+hz1+10)>609.6) THEN !found the 2000ft level
130  z2=fis(i,j)*gi+609.6
131  rt=(z2-h(i,j,lp))/(h(i,j,lp-1)-h(i,j,lp))
132  u2=u(i,j,lp)+rt*(u(i,j,lp-1)-u(i,j,lp))
133  v2=v(i,j,lp)+rt*(v(i,j,lp-1)-v(i,j,lp))
134  k2=lp
135  exit
136  END IF
137  END DO
138  END IF
139 
140 !computer vector difference
141  llws(i,j) = spval
142  if(u10(i,j)<spval.and.v10(i,j)<spval) &
143  llws(i,j)=sqrt((u2-u10(i,j))**2+(v2-v10(i,j))**2)/ &
144  609.6 * 1.943*609.6 !unit: knot/2000ft
145  ENDDO
146 
147 100 CONTINUE
148 
149  RETURN

181 ! OMGA - Vertical velocity (Pa/sec)
182 !
183 ! OUTPUT ARGUMENT LIST:
184 ! ICING - ICING CONDITION (1 or 0)
185 !
186 ! OUTPUT FILES:
187 ! NONE
188 !
189 ! SUBPROGRAMS CALLED:
190 ! UTILITIES:
191 ! LIBRARY:
192 ! NONE
